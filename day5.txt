using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter a number: ");
        int number = int.Parse(Console.ReadLine());
        long factorial = Factorial(number);
        Console.WriteLine($"The factorial of {number} is {factorial}");
    }

    static long Factorial(int n)
    {
        if (n <= 1)
            return 1;
        else
            return n * Factorial(n - 1);
    }
}

		c.   Find out sum of alternate digits of the given 5 digit number
					Input :   12345
					Output :     1 + 3 + 5  =  9 

using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter a 5-digit number: ");
        string input = Console.ReadLine();

        if (input.Length == 5 && int.TryParse(input, out int number))
        {
            int sum = SumOfAlternateDigits(input);
            Console.WriteLine($"The sum of alternate digits is: {sum}");
        }
        else
        {
            Console.WriteLine("Please enter a valid 5-digit number.");
        }
    }

    static int SumOfAlternateDigits(string numberStr)
    {
        int sum = 0;
        for (int i = 0; i < numberStr.Length; i += 2)
        {
            sum += int.Parse(numberStr[i].ToString());
        }
        return sum;
    }
}

		d.   Given two numbers a and b, the task is to find the GCD of the two numbers.

using System;

class Program
{
    static void Main()
    {
        Console.Write("Enter the first number: ");
        int a = int.Parse(Console.ReadLine());

        Console.Write("Enter the second number: ");
        int b = int.Parse(Console.ReadLine());

        int gcd = FindGCD(a, b);
        Console.WriteLine($"The GCD of {a} and {b} is {gcd}");
    }

    static int FindGCD(int a, int b)
    {
        while (b != 0)
        {
            int temp = b;
            b = a % b;
            a = temp;
        }
        return a;
    }
}
